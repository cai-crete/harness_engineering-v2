// lib/buildSystemPrompt.ts
export function buildSystemPrompt(
  principleProtocol: string,
  knowledgeDocs: string[] = []
): string {
  if (!principleProtocol) {
    throw new Error("Principle Protocol is required");
  }
  return [principleProtocol, ...knowledgeDocs].join("\n\n---\n\n");
}

// app/api/generate/route.ts
import { GoogleGenAI } from "@google/genai";
import { buildSystemPrompt } from "@/lib/buildSystemPrompt";

const ai = new GoogleGenAI({ apiKey: process.env.GEMINI_API_KEY });

export async function POST(request: Request) {
  try {
    const { userInput } = await request.json();

    if (!userInput || typeof userInput !== "string") {
      return Response.json({ error: "Invalid input" }, { status: 400 });
    }

    const systemPrompt = buildSystemPrompt(
      principleProtocol,    // import from protocol file
      [knowledgeDoc1, ...]  // import from knowledge doc files
    );

    const response = await ai.models.generateContent({
      model: "gemini-2.5-pro",
      config: { systemInstruction: systemPrompt },
      contents: [{ role: "user", parts: [{ text: userInput }] }],
    });

    return Response.json({
      result: response.text,
      usageMetadata: response.usageMetadata,
    });

  } catch (error) {
    console.error("API Error:", error);
    return Response.json(
      { error: "Generation failed. Please try again." },
      { status: 500 }
    );
  }
}

========================================
PATTERN 4: RETRY WITH EXPONENTIAL BACKOFF
========================================

async function callWithRetry(
  fn: () => Promise<any>,
  maxRetries = 2
): Promise<any> {
  for (let i = 0; i <= maxRetries; i++) {
    try {
      return await fn();
    } catch (error: any) {
      if (i === maxRetries) throw error;
      // 429 = quota exceeded, 503 = service unavailable
      if (error.status === 429 || error.status === 503) {
        await new Promise(r => setTimeout(r, Math.pow(2, i) * 1000));
        continue;
      }
      throw error;
    }
  }
}

========================================
MODEL REFERENCE
========================================

Model ID          | Use case
------------------|------------------------------------------
gemini-2.5-pro    | Default for all CAI nodes (highest quality)
gemini-2.5-flash  | Development/testing (faster, lower cost)

Always use gemini-2.5-pro for production Protocol execution.

========================================
SECURITY CHECKLIST
========================================

[ ] GEMINI_API_KEY in .env.local only
[ ] Protocol content served from server side only (never exposed to client)
[ ] systemInstruction always populated (never null/undefined)
[ ] Input validation: type check + length limit before API call
[ ] Error messages to user: generic (no internal details)
[ ] Image upload: validate file type and size before processing
